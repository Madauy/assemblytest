DATA:
  arr 7
      12
      4
      0
  len 4
  one 1

CODE:
  MOV B, arr    ; B = dirección base del arreglo
  MOV A, (len)  ; A = longitud del arreglo

MIN_MAX_SORT:
  CMP A, 0
  JLE END_SORT

  PUSH A
  PUSH B

  MOV A, (B)    ; Guardar mínimo en A
  MOV B, (B)    ; Guardar máximo en B

MIN_LOOP:
  INC (len)
  CMP (len), A
  JGE UPDATE_MIN

  MOV A, (len)
  JMP CONTINUE_MIN_LOOP

UPDATE_MIN:
  CMP (len), B
  JLE CONTINUE_MIN_LOOP

  MOV B, (len)

CONTINUE_MIN_LOOP:
  ADD (len), (one)
  CMP (len), (arr)
  JNE MIN_LOOP

  MOV (arr), A
  MOV (len), B

  POP B
  POP A
  SUB A, (one)
  JMP MIN_MAX_SORT

END_SORT:
