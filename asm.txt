DATA:
    arr 7
        12
        4
        0
    len 4

CODE:
    MOV B, arr
    MOV A, (len)

MIN_MAX_SORT:
    DEC A
    JEQ END_SORT
    PUSH A
    MOV A, B
    ADD A, len
    DEC A
    PUSH A

MIN_LOOP:
    MOV A, (B)
    CMP A, (B+1)
    JGT MIN_SWAP

NEXT_ELEMENT:
    INC B
    POP A
    CMP A, B
    JNE MIN_LOOP
    PUSH A

    POP A
    JMP MIN_MAX_SORT

MIN_SWAP:
    MOV A, (B+1)
    MOV (B+1), (B)
    MOV (B), A
    JMP NEXT_ELEMENT

END_SORT:
