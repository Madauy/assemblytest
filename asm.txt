DATA:
    arr 7
        12
        4
        0
    len 4

CODE:
    MOV B, arr    ; B = dirección base del arreglo
    MOV A, (len)  ; A = longitud del arreglo

MIN_MAX_SORT:
    DEC A
    CMP A, 0
    JLE END_SORT
    PUSH A

    MOV A, B      ; A = dirección base del arreglo
    PUSH A

    ADD A, B
    SUB A, 1
    MOV B, A      ; B = dirección del último elemento del arreglo

MIN_LOOP:
    CMP A, B
    JGT SWAP
    DEC B
    JMP MIN_LOOP

SWAP:
    MOV B, A
    ADD A, 1
    PUSH A
    MOV A, (B)
    ADD A, (B)
    MOV B, A
    POP A
    SUB A, 1
    MOV B, A

    POP A
    INC A
    JMP MIN_LOOP

END_SORT:
    POP A
    INC A
    MOV B, A
    JMP MIN_MAX_SORT
