DATA:
  arr 7
      12
      4
      0
  len 4
  one 1
  temp 0

CODE:
  MOV B, arr    ; B = dirección base del arreglo
  MOV A, (len)  ; A = longitud del arreglo

MIN_MAX_SORT:
  CMP A, 0
  JLE END_SORT

  PUSH A
  PUSH B

  MOV A, (B)    ; Guardar mínimo en A
  MOV B, (B)    ; Guardar máximo en B
  MOV A, (len)  ; Preparar el contador del bucle
  MOV (temp), A

MIN_LOOP:
  MOV A, (temp)
  ADD A, (one)
  MOV (temp), A
  CMP A, (arr)
  JNE CONTINUE_MIN_LOOP

  JMP END_MIN_LOOP

CONTINUE_MIN_LOOP:
  CMP A, (arr)
  JGE UPDATE_MIN

  MOV A, (arr)

  JMP NEXT_ITERATION_MIN_LOOP

UPDATE_MIN:
  CMP A, (arr)
  JLE NEXT_ITERATION_MIN_LOOP

  MOV B, (arr)

NEXT_ITERATION_MIN_LOOP:
  ADD A, (one)
  MOV (temp), A
  JMP MIN_LOOP

END_MIN_LOOP:
  MOV (arr), A
  MOV (len), B

  POP B
  POP A
  SUB A, (one)
  JMP MIN_MAX_SORT

END_SORT:
